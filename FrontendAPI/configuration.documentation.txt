Each Script has:
-name
-configuration:
  -interfaces:
    -standard, name and conf(Conf is the parameter name)
  -input:
      -type: text,textArea, range or number
      -conf
      -unit
      -name
      -standard
      -max,min, step (Only number and range)
      -cols, rows (Only type textArea)
-graph:
    -type: histogram or line
    -titel
    -travers: List of travers objects
    	-x (Value name)
	-title (title of the travers)
    	-y (Value name, only line)
    -size (Histogram size)
    -id (Must be unique)
    -max (Maximal datapoints)
